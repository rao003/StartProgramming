CREATE TABLE `vhs`.`users` ( `id` INT UNSIGNED NOT NULL AUTO_INCREMENT , `email` VARCHAR(255) NOT NULL , `pasword` VARCHAR(255) NOT NULL , `name` VARCHAR(255) NOT NULL , `lastname` VARCHAR(255) NOT NULL , `created` TIMESTAMP NOT NULL DEFAULT CURRENT_TIMESTAMP , `updated` TIMESTAMP NULL DEFAULT NULL , PRIMARY KEY (`id`), UNIQUE `text` (`email`)) ENGINE = InnoDB;
